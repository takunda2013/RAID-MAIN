   - last_login_at: TEXT, NULL

   - id: INTEGER, PRIMARY KEY, AUTOINCREMENT

   - action: TEXT, NOT NULL
   - target_user_id: INTEGER, NULL, FOREIGN KEY -> users.id
   - detail: TEXT, NULL
   - created_at: TEXT, NOT NULL (ISO datetime)

3) documents
   - id: INTEGER, PRIMARY KEY, AUTOINCREMENT
   - user_id: INTEGER, NOT NULL, FOREIGN KEY -> users.id ON DELETE CASCADE
   - original_filename: TEXT, NOT NULL
   - stored_path: TEXT, NOT NULL
   - report_path: TEXT, NULL
   - status: TEXT, NOT NULL (CHECK: 'uploaded' | 'processed')
   - demo_scenario: TEXT, NOT NULL
   - label: TEXT, NULL
   - confidence: INTEGER, NULL
   - ai_probability: INTEGER, NULL
   - word_count: INTEGER, NOT NULL, DEFAULT 0
   - uploaded_at: TEXT, NOT NULL
   - processed_at: TEXT, NULL

Relationships and cardinality:
- `users` 1 — 0..* `documents` (via `documents.user_id`; ON DELETE CASCADE)
- `users` 0..1 — 0..* `audit_log` as actor (`audit_log.actor_user_id` nullable)
